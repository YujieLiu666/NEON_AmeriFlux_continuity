# IQR filtering
{
  filter_extremes <- function(x) {
  Q1 <- quantile(x, 0.25, na.rm = TRUE)
  Q3 <- quantile(x, 0.75, na.rm = TRUE)
  IQR_value <- Q3 - Q1
  lower_bound <- Q1 - 1.5 * IQR_value
  upper_bound <- Q3 + 1.5 * IQR_value
  x[x < lower_bound | x > upper_bound] <- NA  # Set extreme values to NA
  return(x)
}

df$MO_LENGTH_NEON <- filter_extremes(df$MO_LENGTH_NEON)
